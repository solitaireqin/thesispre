\section{Background \& Motivation}
%------------------------------------------------
\begin{frame}{Background \& Motivation}
    \begin{itemize}
        \item \textbf{Rise of Embodied AI}:
        \begin{itemize}
            \item Vision-Language-Action (VLA) models directly map high-dimensional semantic instructions and visual observations to low-level spatial actions.
        \end{itemize}
        \vspace{0.3cm}
        \item \textbf{Core Challenges in Current Methods}:
        \begin{itemize}
            \item \textbf{Sim-to-Real Gap}: Over-reliance on simulation pre-training leads to drastic drops in real-world deployment success rates.
            \item \textbf{Bimanual Coordination}: Continuous dual-arm manipulation demands extremely high spatial and temporal synchronization.
            \item \textbf{Inference Latency}: Large models with billions of parameters suffer from severe computational bottlenecks, causing physical mechanical jitter.
        \end{itemize}
        \vspace{0.3cm}
        \item \textbf{Objective}:
        \begin{itemize}
            \item Bypass conventional simulation-based pre-training entirely.
            \item Build and deploy a lightweight, end-to-end bimanual VLA pipeline \textbf{directly in a real-world physical environment}.
        \end{itemize}
    \end{itemize}
\end{frame}

\section{Core Task: Bimanual Apple Transfer}
%------------------------------------------------
\begin{frame}{Core Task: Bimanual Apple Transfer}
    This task requires \textbf{sub-millimeter spatial precision} and strict \textbf{coordination logic}. It is decomposed into four highly time-dependent phases:
    
    \begin{figure}
        \centering
        \includegraphics[width=0.85\textwidth]{figures/3.2.png}
        \caption{Sequential Execution Phases of Bimanual Handover}
    \end{figure}
    
    \begin{itemize}
        \item \textbf{Phase 1}: Left arm acquires the target object.
        \item \textbf{Phase 2}: Delivers to the center desk and retracts to avoid collision.
        \item \textbf{Phase 3}: Right arm executes relay grasp (millimeter-level tolerance).
        \item \textbf{Phase 4}: Smooth placement into the target receptacle.
    \end{itemize}
\end{frame}

\section{VLA Flow-Matching Architecture}
%------------------------------------------------
\begin{frame}{Continuous Flow-Matching VLA ($\pi_0$ \& $\pi_{0.5}$ Arch)}
    \begin{columns}
        \begin{column}{0.6\textwidth}
            \begin{itemize}
                \item Adopted the state-of-the-art 3B-parameter $\pi_0$ Flow-Matching representation.
                \item Evaluated and fine-tuned the upgraded $\pi_{0.5}$ model, integrating dense Quantile-normalized Action Chunking.

\section{Asynchronous Deployment \& Latency}
%------------------------------------------------
\begin{frame}{Low-Latency Asynchronous Deployment}
    To resolve the severe latency bottlenecks of generative inference in high-frequency physical closed loops:
    \begin{itemize}
        \item Engineered an \textbf{Asynchronous Client-Server Inference Architecture}.
        \item Decoupled heavy visual flow-matching from physical execution loops.
        \item \textbf{EMA Filtering}: Applied Exponential Moving Average smoothing to buffer overlapping action chunks.
    \end{itemize}
    
    \begin{figure}
        \centering
        \includegraphics[width=0.6\textwidth]{figures/inference_latency.png}
        \caption{Inference \& Kinematic Control Latency Profiling}
    \end{figure}
    Result: Execution cycle instruction latency is successfully constrained to \textbf{< 50 ms}, eliminating mechanical divergence and destructive jitter.
\end{frame}

\section{Conclusion \& Future Work}
%------------------------------------------------
\begin{frame}{Conclusion \& Future Work}
    \begin{itemize}
        \item \textbf{Summary of Contributions}:
        \begin{itemize}
            \item Built, aligned, and deployed an end-to-end, simulation-free bimanual VLA pipeline directly in the real world.
            \item Empirically validated that \textbf{extremely small-scale (200 trials) but highly synchronized real-world data} dramatically outperforms massive generic datasets in fine-tuning VLA operations.
            \item Overcame the "generative inference latency" bottleneck using an asynchronous execution framework combined with $\pi_{0.5}$ chunking.
        \end{itemize}
        \vspace{0.4cm}
        \item \textbf{Future Work}:
        \begin{itemize}
            \item \textbf{Dynamic Interference}: Evaluate system robustness against random human physical perturbations during handovers.
            \item \textbf{Multi-Modal Component}: Incorporate tactile/force feedback to mitigate spatial judgment errors caused by visual occlusions.
        \end{itemize}
    \end{itemize}
\end{frame}
