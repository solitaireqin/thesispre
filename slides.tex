\section{研究背景与动机}
%------------------------------------------------
\begin{frame}{研究背景与动机 (Background \& Motivation)}
    \begin{itemize}
        \item \textbf{具身智能的兴起 (Embodied AI)}：
        \begin{itemize}
            \item 视觉-语言-动作模型 (VLAs) 能够将高维的语义指令和视觉图像直接映射为底层空间动作，是目前具身智能的重要技术路线。
        \end{itemize}
        \vspace{0.3cm}
        \item \textbf{当前面临的核心挑战}：
        \begin{itemize}
            \item \textbf{仿真与物理的鸿沟 (Sim-to-real Gap)}：传统方案过度依赖仿真预训练，导致真实物理部署成功率极低。
            \item \textbf{双臂协同的指数级复杂度 (Bimanual Coordination)}：连续双臂操作在空间与时间上的同步要求极高。
            \item \textbf{实时推理延迟 (Inference Latency)}：拥有数十亿参数的大模型在生成轨迹时由于算力瓶颈极易造成物理机械抖动。
        \end{itemize}
        \vspace{0.3cm}
        \item \textbf{本论文的核心目标}：
        \begin{itemize}
            \item 完全绕过传统的仿真预训练，\textbf{直接在纯物理世界真实环境中}，构建并部署轻量级、端到端的双臂 VLA 操作流水线。
        \end{itemize}
    \end{itemize}
\end{frame}

    \begin{columns}

\section{核心任务：双臂物资接力与传递}
%------------------------------------------------
\begin{frame}{核心任务设计：双臂苹果传递 (Bimanual Apple Transfer)}
    该任务需要\textbf{亚毫米级的空间精度验证}与严谨的\textbf{交接逻辑协同}。整个任务被划解为四个具有高度时间依赖性的逻辑阶段：
    
    \begin{figure}
        \centering
        \includegraphics[width=0.85\textwidth]{../report/figures/3.2.png}
        \caption{双臂交接传递任务分镜序列图}
    \end{figure}
    
    \begin{itemize}
        \item (b) \textbf{阶段1}：左臂获取目标物体。
        \item (c) \textbf{阶段2}：运送至中心区交接并让位，防止相撞。
        \item (d) \textbf{阶段3}：右臂接力抓取，要求空间偏移量在毫米级别之内。
        \item (e) \textbf{阶段4}：稳态运送至右侧目标靶位。
    \end{itemize}
\end{frame}

\section{算法模型与流匹配架构}
%------------------------------------------------
\begin{frame}{连续流匹配 VLA 基础模型 ($\pi_0$ \& $\pi_{0.5}$ Arch)}
    \begin{columns}
        \begin{column}{0.6\textwidth}
            \begin{itemize}
                \item 本研究选用了前沿的30亿规模参数的 $\pi_0$ 视觉-语言-动作流匹配 (Flow-Matching) 架构。
                \item 测试并微调了升级版 $\pi_{0.5}$ 模型结构，融合了密集分位数归一化动作切块 (Action Chunking)。
                \item \textbf{主时钟对齐}：使用严格的“全局时间主时钟（Temporal Master Clock）”解决高配摄像头与机械臂指令通道的频率乱序错位。
                \item 最终收集利用了仅含 200 组真实且高保真的双臂成功演示进行了物理数据微调。
            \end{itemize}
        \end{column}

\section{实验部署与端到端延迟控制}
%------------------------------------------------
\begin{frame}{异步部署与低延迟攻坚 (Low-Latency Asynchronous Deployment)}
    针对 VLA 生成式轨迹推理在高频物理闭环存在的严重延迟瓶颈：
    \begin{itemize}
        \item 设计并应用了\textbf{异步客户端-服务器推理机制 (Asynchronous Client-Server Architecture)}。
        \item 将沉重的视觉计算负荷与底层轨迹运行控制进行并行解耦。
        \item 实验表明：此框架有效把闭环控制周期压缩在 \textbf{<50 ms}，消除了执行器机械发散与破坏性颤动。
    \end{itemize}

\section{总结与展望}
%------------------------------------------------
\begin{frame}{总结与展望 (Conclusion \& Future Work)}
    \begin{itemize}
        \item \textbf{核心贡献总结}：
        \begin{itemize}
            \item 从零搭建、对齐、并部署了一套全链路不依赖仿真的双臂具身操作大模型应用框架。
            \item 实验论证了本系统中的一个核心论点：\textbf{极小尺度但“绝对时序高频对齐”的实采数据，在精调流匹配类 VLA 模型时表现优于庞大的低质数据集}。
            \item 通过引入异步并发机制搭配 $\pi_{0.5}$，跨越了长期困扰视觉操作端到端链路的“生成式推理高延迟”问题。
        \end{itemize}
        \vspace{0.4cm}
        \item \textbf{未来研究展望}：
        \begin{itemize}
            \item \textbf{动态干扰}：探讨模型在执行交接过程中遭受人为随机外力干涉恢复情况。
            \item \textbf{多模态融合}：未来在此纯视觉链路中引入触觉传感融合，缓解因相机视角遮挡导致的空间位域判断误差。
        \end{itemize}
    \end{itemize}
\end{frame}
