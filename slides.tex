% ====== Frame 4: Background II ======
\begin{frame}{Introduction: The Rise of VLA Models}
    {\Large \textbf{Vision-Language-Action Architectures}}
    \vspace{0.5cm}
    \begin{itemize}
        \item \textbf{Semantic Grounding:} VLMs (like GPT-4V) understand the world but cannot physically interact. 
        \item \textbf{Action Injection:} VLA models inject physical actuation tokens directly into the generative stream.
        \item \textbf{Implementation:} Translates natural language (``pick up the apple'') and RGB streams into multi-dimensional joint velocity coordinates.
    \end{itemize}

% ====== Frame 5: Background III ======
\begin{frame}{Core Challenges: The Deployment Gap}
    {\Large \textbf{Why Foundational AI Fails in the Physical World}}
    \vspace{0.5cm}
    \begin{itemize}
        \item \textbf{The Sim-to-Real Disconnect:} Pre-training on massive simulated datasets ignores strict hardware tolerances and physical friction, causing disastrous physical deployment drops.
        \item \textbf{Generic Priors:} "Internet-scale" embodied datasets (e.g., Open X-Embodiment) provide macroscopic context but lack the sub-millimeter geometry calibration demanded by industrial tasks.
    \end{itemize}
\end{frame}
