% ====== Frame 3: Introduction 1 ======
\begin{frame}{Introduction: The Evolution of Embodied AI}
    \begin{itemize}
        \item \textbf{Paradigm Shift in Robotics}: The integration of Vision-Language-Action (VLA) models has fundamentally transformed robotics from rigid, modular pipelines (perception $\rightarrow$ planning $\rightarrow$ control) to end-to-end continuous learning architectures.
        \item \textbf{Direct Behavioral Mapping}: VLAs process high-dimensional multi-modal inputs (e.g., natural language instructions and RGB continuous streams) to directly yield low-level spatial joint actuations.
        \item \textbf{Industrial Relevance}: Robust bimanual manipulation is mandatory for unstructured industrial and logistical environments, yet it remains critically underexplored due to exponential coordination complexities.
    \end{itemize}
\end{frame}

% ====== Frame 4: Introduction 2 ======
\begin{frame}{Core Challenges: The Deployment Bottlenecks}
    \begin{itemize}
        \item \textbf{The Sim-to-Real Gap}: Contemporary VLA frameworks (e.g., OpenVLA, Octo) heavily rely on macroscopic semantic priors derived from Internet-scale or simulated datasets. These priors fail catastrophically during high-precision, sub-millimeter physical interactions.
        \item \textbf{Dual-Arm Interdependency}: Extending standard manipulation to bimanual scenarios induces a combinatorial explosion in the configuration space, requiring absolute spatio-temporal synchronization to prevent collisions or logic breakdown.
        \item \textbf{Inference Latency vs. Kinematic Stability}: Multi-billion parameter generative models exhibit significant inference delays (e.g., 100ms - 500ms). When injected directly into high-frequency physical control loops, this latency manifests as destructive mechanical divergence and jitter.
    \end{itemize}
\end{frame}

% ====== Frame 5: Objectives ======
\begin{frame}{Research Objectives and Contributions}
    \begin{itemize}
        \item \textbf{Objective 1 (System Architecture)}: Construct an end-to-end, simulation-free bimanual VLA pipeline operating securely within the constraints of real-world physical dynamics.
        \item \textbf{Objective 2 (Data Protocol)}: Establish a zero-blind-spot spatial data collection methodology to yield ultra-high-fidelity, temporally localized demonstrations, bypassing the reliance on massive generic datasets.
        \item \textbf{Objective 3 (Algorithmic Fine-Tuning)}: Empirically validate the adaptiveness of novel continuous Flow-Matching architectures ($\pi_0$ and $\pi_{0.5}$) for highly correlated 14-DoF control.
        \item \textbf{Objective 4 (Asynchronous Deployment)}: Engineer a decoupled Client-Server architecture and Exponential Moving Average (EMA) filtering to constrain execution cycle latency beneath physiological mechanical limits.
    \end{itemize}
\end{frame}

\begin{frame}{Methodology: Actuation and Teleoperation Infrastructure}

    \begin{itemize}
        \item \textbf{Kinematic Configuration}: Dual AgileX Piper robotic manipulators (12 DoF) coupled with symmetrical 1-DoF AgileX Pika Grippers, yielding a unified 14-DoF optimization target.
        \item \textbf{Spatial Teleoperation Setup}: Utilized the AgileX Pika Data Suit integrated with four Pika Station base stations arrayed in a $45^\circ$ adversarial geometry to strictly eliminate positional calculation blockages.
        \item \textbf{Sampling Rate Governance}: Hand-pose kinematics and gripper trigger states were structurally mapped and hard-capped at 50 Hz to align with the underlying CAN bus physiological bandwidth.
    \end{itemize}
\end{frame}

% ====== Frame 8: Task Definition ======
\begin{frame}{Task Formulation: Bimanual Apple Transfer}
    We formalized a dynamic bimanual material-handling task serving as a rigorous evaluation primitive for visual-motor alignment:
    \begin{itemize}
        \item \textbf{Semantic Definition}: ``Use the left arm to pick up the apple from the blue board, place it in the center of the desk, then use the right arm to grasp it, and drop it into the green bowl.''
        \item \textbf{Geometric Constraints}: The smooth, unstructured surface of the target object combined with the exceptionally narrow geometric clearance of the Pika grippers demands rigorous millimeter-level spatial and rotational accuracy.
        \item \textbf{Critical Relay Phase}: The right manipulator must extract the target strictly from its deposited state without shifting its center of mass; spatial misalignment induces immediate cascading collision.
    \end{itemize}
\end{frame}

% ====== Frame 9: Task Phases ======
\begin{frame}{Task Formulation: Temporal Execution Logic}
    The overarching semantic instruction is functionally unrolled into four distinct, temporally dependent Markov decision phases:
    \begin{itemize}
        \item \textbf{Phase 1 (Object Acquisition)}: Left arm calculates geometric vector and lifts the target.
        \item \textbf{Phase 2 (Centering \& Evasion)}: Navigation to the workspace center, deposition, and required structural retraction to clear spatial collision blocks.
        \item \textbf{Phase 3 (Relay Extraction)}: Initiated conditionally based on visual static verification; right arm secures a secondary bounding box.
        \item \textbf{Phase 4 (Target Placement)}: Right arm performs guided traversal to the target receptacle and smoothly releases the payload.
    \end{itemize}

% ====== Frame 10: Data Protocol ======
\begin{frame}{Data Protocol: Spatio-Temporal High-Fidelity Engineering}
    \begin{itemize}
        \item \textbf{Demonstration Cardinality}: Rather than relying on thousands of noisy simulations, we collected precisely \textbf{200 optimal bimanual demonstrations} administered by expert operators.
        \item \textbf{Temporal Master Clock Synchronization}: A critical engineering intervention. Visual inputs (30 Hz) and proprioceptive metrics (50 Hz) intrinsically misalign. We anchored all data modalities to a rigid absolute Master Timestamp.
        \item \textbf{Scientific Hypothesis}: This asserts that superior temporal correlation and zero-occlusion data engineering intrinsically outweigh raw dataset volume when tuning foundational VLAs.
    \end{itemize}
\end{frame}

% ====== Frame 12: VLA Architecture (pi0.5) ======
\begin{frame}{Algorithmic Enhancement: The $\pi_{0.5}$ Topology}
    To combat single-frame sensory noise and amplify temporal reasoning, we integrated the $\pi_{0.5}$ architectural upgrade:
    \begin{itemize}
        \item \textbf{Action Chunking Integration}: Instead of predicting instantaneous $t+1$ states, the network forecasts an overlapping sequence of prospective temporal actions (\textit{chunks}).
        \item \textbf{Dense Quantile Normalization}: The flow matching processes are stabilized using quantile conditioning across the chunk horizons, mapping highly heterogeneous joint distributions efficiently.
        \item \textbf{Fine-Tuning Objective}: Models were tuned using direct L1 Flow Matching error against the 200 localized trajectory benchmarks. 
    \end{itemize}
\end{frame}

% ====== Frame 13: Deployment Constraints ======
\begin{frame}{Deployment Conundrum: Generative Inference Latency}
    \begin{itemize}
        \item \textbf{Computational Burden}: Generating flow-matched vectors sequentially through a 3B parameter Vision-Language model on standard RTX 4090 architecture induces inference delays reaching \textbf{100ms -- 300ms}.
        \item \textbf{The Frequency Mismatch}: Low-level physical PI/PD motion controllers expect command refreshes at $>20$ Hz ($\le 50$ms).
        \item \textbf{Physical Consequence}: Subjecting high-torque actuations to such sparse and delayed action updates provokes catastrophic mechanical resonance, invalidating the generative output entirely.
    \end{itemize}
\end{frame}

% ====== Frame 14: Asynchronous Client-Server ======
\begin{frame}{System Optimization: Asynchronous Execution \& EMA}
    To dismantle the generative latency barrier, we designed a custom deployment framework:
    \begin{itemize}
        \item \textbf{Asynchronous Client-Server Network}: We structurally severed the compute-heavy visual embedding processes from the raw kinematic execution loop. The Server computes $\pi_{0.5}$ chunks in the background while the Client streams localized joint states concurrently.
        \item \textbf{EMA Temporal Smoothing}: Exponential Moving Average filters were applied over the overlapping predictions. As newer chunks arrive, their distributions smoothly average into the ongoing execution sequence seamlessly.
    \end{itemize}
\end{frame}
